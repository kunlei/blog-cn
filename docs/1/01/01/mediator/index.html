Mediator Pattern

 Posted on January 1, 0001  |  3 minutes  |  442 words  |  Kunlei Lian

Mediator Pattern is one of the behavioral patterns and it aims to decouple some related objects by using a separate object to manage their communications. This may better be explained by an example. Think of the stock trading system as a mediator to manage buying and selling of stocks. Individual investors do not talk to other tradres directly, instead they interact with the trading system directly.

Let’s define the mediator interface as below.

1public interface Mediator {
2    void addTrader(Trader trader);
3    void buy(double amount);
4    void sell(double amount);
5}

and the trading system implements the interface:

 1public class Market implements Mediator {
 2    private List<Trader> traders;
 3    private double value;
 4
 5    public Market() {
 6        traders = new ArrayList<>();
 7        value = 10000;
 8    }
 9
10    @Override
11    public void addTrader(Trader trader) {
12        traders.add(trader);
13    }
14
15    @Override
16    public void buy(double amount) {
17        value -= amount;
18    }
19
20    @Override
21    public void sell(double amount) {
22        value += amount;
23    }
24
25    @Override
26    public String toString() {
27        return "Market{" +
28                "value=" + value +
29                '}';
30    }
31}

Now we have investors ready to trade in the market:

 1public class Trader {
 2    protected Mediator mediator;
 3    protected double cash;
 4    protected double stock;
 5
 6    public Trader(Mediator mediator, double cash) {
 7        this.mediator = mediator;
 8        this.cash = cash;
 9        stock = 0;
10    }
11
12    void buy(double amount) {
13        if (amount > cash) {
14            System.out.println("You don't have that much money!");
15            return;
16        }
17        mediator.buy(amount);
18        cash -= amount;
19        stock += amount;
20    }
21
22    void sell(double amount) {
23        if (amount > stock) {
24            System.out.println("You don't have that much stock to sell!");
25            return;
26        }
27        mediator.sell(amount);
28        cash += amount;
29        stock -= amount;
30    }
31
32    @Override
33    public String toString() {
34        return "Trader{" +
35                "cash=" + cash +
36                ", stock=" + stock +
37                '}';
38    }
39}

Let the game begin:

 1public class TradingSystem {
 2
 3    public static void main(String[] args) {
 4        Market market = new Market();
 5        Trader xiaoMing = new Trader(market, 1000);
 6        Trader gouDan = new Trader(market, 500);
 7        Trader sanMao = new Trader(market, 1300);
 8
 9        market.addTrader(xiaoMing);
10        market.addTrader(gouDan);
11        market.addTrader(sanMao);
12
13        xiaoMing.buy(200);
14        gouDan.buy(300);
15        sanMao.buy(500);
16        System.out.println(market.toString());
17        System.out.println(xiaoMing.toString());
18        System.out.println(gouDan.toString());
19        System.out.println(sanMao.toString());
20
21        gouDan.sell(100);
22        sanMao.sell(200);
23        System.out.println();
24        System.out.println(market.toString());
25        System.out.println(xiaoMing.toString());
26        System.out.println(gouDan.toString());
27        System.out.println(sanMao.toString());
28    }
29}

4Trader{cash=800.0, stock=500.0}
5
6Market{value=9300.0}
7Trader{cash=800.0, stock=200.0}
8Trader{cash=300.0, stock=200.0}
9Trader{cash=1000.0, stock=300.0}
